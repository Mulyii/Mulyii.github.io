莫队

  • 算法原理:
    对于序列上的查询问题, 如果[l, r] 的答案可以扩展到[l- 1, r], [l + 1, r], [l, r-1], [l, r + 1], 那么对于所有询问, 离线处理后可以在的时间内找到所有问题的答案.

  • 排序方法, 以及具体实现
    对于区间, 以所在块的编号为第一关键字升序, 为第二关键字排序升序.
    优化: 若所在块编号为奇数, 则按为第二关键字升序, 反正则按降序.
    具体实现见例题, 主要是编写扩展部分的代码

  • 例题:
    Powerful array
    题目大意:
    对于给定数列, 每次询问一个区间, 区间的值是每个数字出现的次数的平方乘这个数本身的和,即

signed main () {
    ios::sync_with_stdio(0), cin.tie(0);
    int n, t;
    cin >> n >> t;
    vector<int> a(n);
    int bs = sqrt(n);
    for(int i = 0; i < n; i ++) {
        cin >> a[i];
    }
    struct query {
        int l, r, id;
    };
    vector<query> q(t);
    for(int i = 0; i < t; i ++) {
        auto &[l, r, id] = q[i];
        cin >> l >> r;
        l --, r --;
        id = i;
    }
    sort(q.begin(), q.end(), [&] (query a, query b) {
        if(a.l / bs != b.l / bs) return a.l < b.l;
        return (a.l / bs) & 1 ? a.r < b.r : a.r > b.r;
    });
    i64 res = 0;
    vector<i64> ans(t);
    vector<int> cnt(MAXN);
    for(int i = 0, ll = 0, rr = -1; i < t; i ++) {
        auto &[l, r, id] = q[i];
        while (rr < r) {
            int x = a[++ rr];
            i64 fi = 1ll * cnt[x] * cnt[x] * x;
            cnt[x] ++;
            i64 se = 1ll * cnt[x] * cnt[x] * x;
            res = res - fi + se;
        }
        while(rr > r) {
            int x = a[rr --];
            i64 fi = 1ll * cnt[x] * cnt[x] * x;
            cnt[x] --;
            i64 se = 1ll * cnt[x] * cnt[x] * x;
            res = res - fi + se;
        }
        while(ll < l) {
            int x = a[ll ++];
            i64 fi = 1ll * cnt[x] * cnt[x] * x;
            cnt[x] --;
            i64 se = 1ll * cnt[x] * cnt[x] * x;
            res = res - fi + se;
        }
        while(ll > l) {
            int x = a[-- ll];
            i64 fi = 1ll * cnt[x] * cnt[x] * x;
            cnt[x] ++;
            i64 se = 1ll * cnt[x] * cnt[x] * x;
            res = res - fi + se;
        }
        ans[id] = res;
    }
    for(i64 i : ans) {
        cout << i << '\n';
    }
}
