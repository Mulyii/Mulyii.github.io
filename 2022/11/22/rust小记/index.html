控制流

if

兼容 C 风格, 可以带()

if x < 5 {
    x += 1;
} else if x < 10 {
    x += 2;
} else {
    x += 3;
}

loop

无限循环体

loop {
    x += 1;
    if x >= 100 {
        break;
    }
}

while

兼容 C 风格, 可以带()

while x < 5 {
    x += 1;
}

for

遍历集合

let a = [0, 1, 2, 3, 4, 5];
for i in a {
    print!("{}", i);
}

逆序遍历索引
for i in (0..5).rev() {
    // --skip--
}

match

类似 C 语言中的switch

enum IpAddr {
    V4(String),
    V6(String),
}
let home = IpAddr::V4(String::from("192, 0, 0, 1"));
let ip = match home {
    IpAddr::V4(s) => {
        s
    },
    IpAddr::V6(s) => s
    // => 表示产生的效果

    other => String::from("0, 0, 0, 0"),
    //match 是穷尽的 
};

弱化版的match

if let IpAddr::V4(s) = home{
    println!("{}", s);
} else {
    cnt += 1;
}

结构体定义与实例

贴个自己写的 BIT 基本就会用了

fn  main () {
    let mut bit = BIT::new(200);
    for i in (1..bit.N) {
        bit.add(i, 1);
    }
    for i in (1..bit.N) {
        println!("{}", bit.sum(i));
    }
}
struct BIT {
    N: i32,
    s: [i32; 1000],
}
impl BIT {
  // 类似构造方法
    fn new(n: i32) -> Self {
        let bit  =  Self {
            N: n,
            s: [0; 1000],
        };
        bit
    }
    fn add(&mut self, x: i32, y: i32) {
        let mut i = x;
        while i < self.N {
            let inx: usize = i.try_into().unwrap(); // 索引要转化成 usize
            self.s[inx] += y;
            i += i & - i;
        }
    }
    fn sum(&self, x: i32) -> i32 {
        let mut res = 0;
        let mut i = x;
        while i > 0 {
            let inx: usize = i.try_into().unwrap();
            res += self.s[inx];
            i -= i & - i;
        }
        res
    }
}

几种常见的封装集合

Vector

定义

let v1 = vec![1, 2, 3, 4];
let v2 = vec![0; 5];
let  mut v3: Vec<i32> = Vec::new();

几个常用的函数

  • push
  • pop
  • append
    // 将 v2 中的元素移动到 v1
v1.append(&mut v2);
  • sort | sort_by
    // 逆序排序
v1.sort_by(|a, b| b.cmp(a));
  • is_empty
  • len
  • resize

String

定义

let s1 = String::from("hello, ");
let s2 = "world!".to_string();
let s3 = s1 + &s2;

Hash Map

定义

let mut map = HashMap::new();
map.insert("red".to_string(), 1);
map.insert("blue".to_string(), 2);
map.insert("red".to_string(), 10);  // 覆盖
map.entry("red".to_string()).or_insert(10); // 不存在则插入一个值
let conut = map.entry("bule".to_string()).or_insert(0);
*conut += 1; // 更新map中的数据
let x = map.get(&"red".to_string()).copied().unwrap_or(0); // 得到一个不可变的值
