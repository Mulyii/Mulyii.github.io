结构体定义与实例

贴个自己写的 BIT 基本就会用了

fn  main () {
    let mut bit = BIT::new(200);
    for i in (1..bit.N) {
        bit.add(i, 1);
    }
    for i in (1..bit.N) {
        println!("{}", bit.sum(i));
    }
}
struct BIT {
    N: i32,
    s: [i32; 1000],
}
impl BIT {
  // 类似构造方法
    fn new(n: i32) -> Self {
        let bit  =  Self {
            N: n,
            s: [0; 1000],
        };
        bit
    }
    fn add(&mut self, x: i32, y: i32) {
        let mut i = x;
        while i < self.N {
            let inx: usize = i.try_into().unwrap(); // 索引要转化成 usize
            self.s[inx] += y;
            i += i & - i;
        }
    }
    fn sum(&self, x: i32) -> i32 {
        let mut res = 0;
        let mut i = x;
        while i > 0 {
            let inx: usize = i.try_into().unwrap();
            res += self.s[inx];
            i -= i & - i;
        }
        res
    }
}

定义
